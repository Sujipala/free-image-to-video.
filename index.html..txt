    <h1>Image to Video</h1>
    <p>Convert a single photo into a 5-second video clip completely free.</p>

    <input type="file" id="imageInput" accept="image/*">
    
    <button id="convertBtn">Generate Video</button>
    
    <div id="status"></div>
    <video id="resultVideo" controls></video>

<script>
    const imageInput = document.getElementById('imageInput');
    const convertBtn = document.getElementById('convertBtn');
    const status = document.getElementById('status');
    const resultVideo = document.getElementById('resultVideo');

    convertBtn.addEventListener('click', async () => {
        const file = imageInput.files[0];
        if (!file) {
            alert('Please select an image first!');
            return;
        }

        convertBtn.disabled = true;
        status.innerText = "Processing image...";

        // 1. Load image into an HTML Image Object
        const img = new Image();
        img.src = URL.createObjectURL(file);
        
        img.onload = function() {
            status.innerText = "Generating video frames...";
            
            // 2. Set up a hidden Canvas (the drawing board)
            const canvas = document.createElement('canvas');
            const ctx = canvas.getContext('2d');
            
            // Set video size to standard 720p HD
            canvas.width = 1280;
            canvas.height = 720;

            // 3. Set up the Canvas Recorder (captures the video stream)
            const stream = canvas.captureStream(30); // 30 frames per second
            const mediaRecorder = new MediaRecorder(stream, { mimeType: 'video/webm' });
            const chunks = [];

            mediaRecorder.ondataavailable = (e) => chunks.push(e.data);
            mediaRecorder.onstop = () => {
                // 5. Video is done! Create a link to play/download it
                const blob = new Blob(chunks, { type: 'video/webm' });
                const videoURL = URL.createObjectURL(blob);
                
                resultVideo.src = videoURL;
                resultVideo.style.display = 'block';
                status.innerText = "Done! Right-click the video to save it.";
                convertBtn.disabled = false;
            };

            // Start recording
            mediaRecorder.start();

            // 4. Draw the image onto the canvas repeatedly over 5 seconds
            let framesCount = 0;
            const totalFrames = 30 * 5; // 30 FPS * 5 Seconds = 150 frames total

            function drawFrame() {
                // Clear canvas and draw background
                ctx.fillStyle = "#000000";
                ctx.fillRect(0, 0, canvas.width, canvas.height);

                // Draw the user's image, scaling it to fit the screen nicely
                const hRatio = canvas.width / img.width;
                const vRatio = canvas.height / img.height;
                const ratio  = Math.min(hRatio, vRatio);
                const shiftX = (canvas.width - img.width * ratio) / 2;
                const shiftY = (canvas.height - img.height * ratio) / 2;  
                
                ctx.drawImage(img, 0, 0, img.width, img.height, shiftX, shiftY, img.width * ratio, img.height * ratio);

                framesCount++;

                if (framesCount < totalFrames) {
                    requestAnimationFrame(drawFrame);
                } else {
                    // Stop recording after 5 seconds
                    mediaRecorder.stop();
                }
            }

            drawFrame();
        };
    });
</script>
